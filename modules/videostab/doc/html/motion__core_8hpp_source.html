82  static RansacParams default2dMotion(MotionModel model)
83  {
84  CV_Assert(model < MM_UNKNOWN);
85  if (model == MM_TRANSLATION)
86  return RansacParams(1, 0.5f, 0.5f, 0.99f);
87  if (model == MM_TRANSLATION_AND_SCALE)
88  return RansacParams(2, 0.5f, 0.5f, 0.99f);
89  if (model == MM_ROTATION)
90  return RansacParams(1, 0.5f, 0.5f, 0.99f);
91  if (model == MM_RIGID)
92  return RansacParams(2, 0.5f, 0.5f, 0.99f);
93  if (model == MM_SIMILARITY)
94  return RansacParams(2, 0.5f, 0.5f, 0.99f);
95  if (model == MM_AFFINE)
96  return RansacParams(3, 0.5f, 0.5f, 0.99f);
97  return RansacParams(4, 0.5f, 0.5f, 0.99f);
98  }
