55 float calcBlurriness(const Mat &frame)
56 {
57  Mat Gx, Gy;
58  Sobel(frame, Gx, CV_32F, 1, 0);
59  Sobel(frame, Gy, CV_32F, 0, 1);
60  double normGx = norm(Gx);
61  double normGy = norm(Gx);
62  double sumSq = normGx*normGx + normGy*normGy;
63  return static_cast<float>(1. / (sumSq / frame.size().area() + 1e-6));
64 }
