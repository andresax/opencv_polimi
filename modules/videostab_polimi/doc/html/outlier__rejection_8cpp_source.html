73 void TranslationBasedLocalOutlierRejector::process(
74  Size frameSize, InputArray points0, InputArray points1, OutputArray mask)
75 {
76  CV_Assert(points0.type() == points1.type());
77  CV_Assert(points0.getMat().checkVector(2) == points1.getMat().checkVector(2));
78 
79  int npoints = points0.getMat().checkVector(2);
80 
81  const Point2f* points0_ = points0.getMat().ptr<Point2f>();
82  const Point2f* points1_ = points1.getMat().ptr<Point2f>();

87  Size ncells((frameSize.width + cellSize_.width - 1) / cellSize_.width,
88  (frameSize.height + cellSize_.height - 1) / cellSize_.height);
89 
90  int cx, cy;
91 
92  // fill grid cells
93

95 
96  for (int i = 0; i < npoints; ++i)
97  {
98  cx = std::min(cvRound(points0_[i].x / cellSize_.width), ncells.width - 1);
99  cy = std::min(cvRound(points0_[i].y / cellSize_.height), ncells.height - 1);
100  grid_[cy * ncells.width + cx].push_back(i);
101  }
102 
103  // process each cell
104 
105  RNG rng(0);
106  int niters = ransacParams_.niters();
107  int ninliers, ninliersMax;
108  vector<int> inliers;
109  float dx, dy, dxBest, dyBest;
110  float x1, y1;
111  int idx;
112 
113  for (size_t ci = 0; ci < grid_.size(); ++ci)
114  {
115  // estimate translation model at the current cell using RANSAC
116 
117  const Cell &cell = grid_[ci];
118  ninliersMax = 0;
119  dxBest = dyBest = 0.f;
120 
121  // find the best hypothesis
122 
123  if (!cell.empty())
124  {
125  for (int iter = 0; iter < niters; ++iter)
126  {
127  idx = cell[static_cast<unsigned>(rng) % cell.size()];
128  dx = points1_[idx].x - points0_[idx].x;
129  dy = points1_[idx].y - points0_[idx].y;
130 
131  ninliers = 0;
132  for (size_t i = 0; i < cell.size(); ++i)
133  {

135  y1 = points0_[cell[i]].y + dy;
136  if (sqr(x1 - points1_[cell[i]].x) + sqr(y1 - points1_[cell[i]].y) <
137  sqr(ransacParams_.thresh))
138  {
139  ninliers++;
140  }
141  }
142 
143  if (ninliers > ninliersMax)
144  {
145  ninliersMax = ninliers;
146  dxBest = dx;
147  dyBest = dy;
148  }
149  }
150  }
151 
152  // get the best hypothesis inliers
153 
154  ninliers = 0;
155  inliers.resize(ninliersMax);
156  for (size_t i = 0; i < cell.size(); ++i)
157  {
158  x1 = points0_[cell[i]].x + dxBest;
159  y1 = points0_[cell[i]].y + dyBest;
160  if (sqr(x1 - points1_[cell[i]].x) + sqr(y1 - points1_[cell[i]].y) <
161  sqr(ransacParams_.thresh))
162  {
163  inliers[ninliers++] = cell[i];
164  }
165  }
166 
167  // refine the best hypothesis
168 
169  dxBest = dyBest = 0.f;
170  for (size_t i = 0; i < inliers.size(); ++i)
171  {
172  dxBest += points1_[inliers[i]].x - points0_[inliers[i]].x;
173  dyBest += points1_[inliers[i]].y - points0_[inliers[i]].y;
174  }
175  if (!inliers.empty())
176  {
177  dxBest /= inliers.size();
178  dyBest /= inliers.size();
179  }
180 
181  // set mask elements for refined model inliers
182 
183  for (size_t i = 0; i < cell.size(); ++i)
184  {
185  x1 = points0_[cell[i]].x + dxBest;
186  y1 = points0_[cell[i]].y + dyBest;
187  if (sqr(x1 - points1_[cell[i]].x) + sqr(y1 - points1_[cell[i]].y) <
188  sqr(ransacParams_.thresh))
189  {
190  mask_[cell[i]] = 1;
191  }
192  else
193  {
194  mask_[cell[i]] = 0;
195  }
196  }
197  }
