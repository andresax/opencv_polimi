127 void ConsistentMosaicInpainter::inpaint(int idx, Mat &frame, Mat &mask)
128 {
129  CV_Assert(frame.type() == CV_8UC3);
130  CV_Assert(mask.size() == frame.size() && mask.type() == CV_8U);
131 
132  Mat invS = at(idx, *stabilizationMotions_).inv();
133  vector<Mat_<float> > vmotions(2*radius_ + 1);
134  for (int i = -radius_; i <= radius_; ++i)
135  vmotions[radius_ + i] = getMotion(idx, idx + i, *motions_) * invS;
136 
137  int n;
138  float mean, var;
139  vector<Pixel3> pixels(2*radius_ + 1);
140 
141  Mat_<Point3_<uchar> > frame_(frame);
142  Mat_<uchar> mask_(mask);
143 
144  for (int y = 0; y < mask.rows; ++y)
145  {
146  for (int x = 0; x < mask.cols; ++x)
147  {
148  if (!mask_(y, x))
149  {
150  n = 0;
151  mean = 0;
152  var = 0;
153 
154  for (int i = -radius_; i <= radius_; ++i)
155  {
156  const Mat_<Point3_<uchar> > &framei = at(idx + i, *frames_);
157  const Mat_<float> &Mi = vmotions[radius_ + i];
158  int xi = cvRound(Mi(0,0)*x + Mi(0,1)*y + Mi(0,2));
159  int yi = cvRound(Mi(1,0)*x + Mi(1,1)*y + Mi(1,2));
160  if (xi >= 0 && xi < framei.cols && yi >= 0 && yi < framei.rows)
161  {
162  pixels[n].color = framei(yi, xi);
163  mean += pixels[n].intens = intensity(pixels[n].color);
164  n++;
165  }
166  }
167 
168  if (n > 0)
169  {
170  mean /= n;
171  for (int i = 0; i < n; ++i)
172  var += sqr(pixels[i].intens - mean);
173  var /= std::max(n - 1, 1);
174 
175  if (var < stdevThresh_ * stdevThresh_)
176  {
177  sort(pixels.begin(), pixels.begin() + n);
178  int nh = (n-1)/2;
179  int c1 = pixels[nh].color.x;
180  int c2 = pixels[nh].color.y;
181  int c3 = pixels[nh].color.z;
182  if (n-2*nh)
183  {
184  c1 = (c1 + pixels[nh].color.x) / 2;
185  c2 = (c2 + pixels[nh].color.y) / 2;
186  c3 = (c3 + pixels[nh].color.z) / 2;
187  }
188  frame_(y, x) = Point3_<uchar>(
189  static_cast<uchar>(c1),
190  static_cast<uchar>(c2),
191  static_cast<uchar>(c3));
192  mask_(y, x) = 255;
193  }
194  }
195  }
196  }
197  }

201 static float alignementError(
202  const Mat &M, const Mat &frame0, const Mat &mask0, const Mat &frame1)
203 {
204  CV_Assert(frame0.type() == CV_8UC3 && frame1.type() == CV_8UC3);
205  CV_Assert(mask0.type() == CV_8U && mask0.size() == frame0.size());
206  CV_Assert(frame0.size() == frame1.size());
207  CV_Assert(M.size() == Size(3,3) && M.type() == CV_32F);
208 
209  Mat_<uchar> mask0_(mask0);
210  Mat_<float> M_(M);
211  float err = 0;
212 
213  for (int y0 = 0; y0 < frame0.rows; ++y0)
214  {
215  for (int x0 = 0; x0 < frame0.cols; ++x0)
216  {
217  if (mask0_(y0,x0))
218  {
219  int x1 = cvRound(M_(0,0)*x0 + M_(0,1)*y0 + M_(0,2));
220  int y1 = cvRound(M_(1,0)*x0 + M_(1,1)*y0 + M_(1,2));
221  if (y1 >= 0 && y1 < frame1.rows && x1 >= 0 && x1 < frame1.cols)
222  err += std::abs(intensity(frame1.at<Point3_<uchar> >(y1,x1)) -
223  intensity(frame0.at<Point3_<uchar> >(y0,x0)));
224  }
225  }
226  }
227 
228  return err;
229 }

232 class MotionInpaintBody
233 {
234 public:
235  void operator ()(int x, int y)
236  {
237  float uEst = 0.f, vEst = 0.f, wSum = 0.f;
238 
239  for (int dy = -rad; dy <= rad; ++dy)
240  {
241  for (int dx = -rad; dx <= rad; ++dx)
242  {
243  int qx0 = x + dx;
244  int qy0 = y + dy;
245 
246  if (qy0 >= 0 && qy0 < mask0.rows && qx0 >= 0 && qx0 < mask0.cols && mask0(qy0,qx0))
247  {
248  int qx1 = cvRound(qx0 + flowX(qy0,qx0));
249  int qy1 = cvRound(qy0 + flowY(qy0,qx0));
250  int px1 = qx1 - dx;
251  int py1 = qy1 - dy;
252 
253  if (qx1 >= 0 && qx1 < mask1.cols && qy1 >= 0 && qy1 < mask1.rows && mask1(qy1,qx1) &&
254  px1 >= 0 && px1 < mask1.cols && py1 >= 0 && py1 < mask1.rows && mask1(py1,px1))
255  {
256  float dudx = 0.f, dvdx = 0.f, dudy = 0.f, dvdy = 0.f;
257 
258  if (qx0 > 0 && mask0(qy0,qx0-1))
259  {
260  if (qx0+1 < mask0.cols && mask0(qy0,qx0+1))
261  {
262  dudx = (flowX(qy0,qx0+1) - flowX(qy0,qx0-1)) * 0.5f;
263  dvdx = (flowY(qy0,qx0+1) - flowY(qy0,qx0-1)) * 0.5f;
264  }
265  else
266  {
267  dudx = flowX(qy0,qx0) - flowX(qy0,qx0-1);
268  dvdx = flowY(qy0,qx0) - flowY(qy0,qx0-1);
269  }
270  }
271  else if (qx0+1 < mask0.cols && mask0(qy0,qx0+1))
272  {
273  dudx = flowX(qy0,qx0+1) - flowX(qy0,qx0);
274  dvdx = flowY(qy0,qx0+1) - flowY(qy0,qx0);
275  }
276 
277  if (qy0 > 0 && mask0(qy0-1,qx0))
278  {
279  if (qy0+1 < mask0.rows && mask0(qy0+1,qx0))
280  {
281  dudy = (flowX(qy0+1,qx0) - flowX(qy0-1,qx0)) * 0.5f;
282  dvdy = (flowY(qy0+1,qx0) - flowY(qy0-1,qx0)) * 0.5f;
283  }
284  else
285  {
286  dudy = flowX(qy0,qx0) - flowX(qy0-1,qx0);
287  dvdy = flowY(qy0,qx0) - flowY(qy0-1,qx0);
288  }
289  }
290  else if (qy0+1 < mask0.rows && mask0(qy0+1,qx0))
291  {
292  dudy = flowX(qy0+1,qx0) - flowX(qy0,qx0);
293  dvdy = flowY(qy0+1,qx0) - flowY(qy0,qx0);
294  }
295 
296  Point3_<uchar> cp = frame1(py1,px1), cq = frame1(qy1,qx1);
297  float distColor = sqr(static_cast<float>(cp.x-cq.x))
298  + sqr(static_cast<float>(cp.y-cq.y))
299  + sqr(static_cast<float>(cp.z-cq.z));
300  float w = 1.f / (sqrt(distColor * (dx*dx + dy*dy)) + eps);
301 
302  uEst += w * (flowX(qy0,qx0) - dudx*dx - dudy*dy);
303  vEst += w * (flowY(qy0,qx0) - dvdx*dx - dvdy*dy);
304  wSum += w;
305  }
306  }
307  }
308  }
309 
310  if (wSum > 0.f)
311  {
312  flowX(y,x) = uEst / wSum;
313  flowY(y,x) = vEst / wSum;
314  mask0(y,x) = 255;
315  }
316  }
317 
318  Mat_<Point3_<uchar> > frame1;
319  Mat_<uchar> mask0, mask1;
320  Mat_<float> flowX, flowY;
321  float eps;
322  int rad;
323 };

339 void MotionInpainter::inpaint(int idx, Mat &frame, Mat &mask)
340 {
341  priority_queue<pair<float,int> > neighbors;
342  vector<Mat> vmotions(2*radius_ + 1);
343 
344  for (int i = -radius_; i <= radius_; ++i)
345  {
346  Mat motion0to1 = getMotion(idx, idx + i, *motions_) * at(idx, *stabilizationMotions_).inv();
347  vmotions[radius_ + i] = motion0to1;
348 
349  if (i != 0)
350  {
351  float err = alignementError(motion0to1, frame, mask, at(idx + i, *frames_));
352  neighbors.push(make_pair(-err, idx + i));
353  }
354  }
355 
356  if (mask1_.size() != mask.size())
357  {
358  mask1_.create(mask.size());
359  mask1_.setTo(255);
360  }
361 
362  cvtColor(frame, grayFrame_, CV_BGR2GRAY);
363 
364  MotionInpaintBody body;
365  body.rad = 2;
366  body.eps = 1e-4f;
367 
368  while (!neighbors.empty())
369  {
370  int neighbor = neighbors.top().second;
371  neighbors.pop();
372 
373  Mat motion1to0 = vmotions[radius_ + neighbor - idx].inv();
374 
375  // warp frame
376 
377  frame1_ = at(neighbor, *frames_);
378 
379  if (motionModel_ != MM_HOMOGRAPHY)
380  warpAffine(
381  frame1_, transformedFrame1_, motion1to0(Rect(0,0,3,2)), frame1_.size(),
382  INTER_LINEAR, borderMode_);
383  else
384  warpPerspective(
385  frame1_, transformedFrame1_, motion1to0, frame1_.size(), INTER_LINEAR,
386  borderMode_);
387 
388  cvtColor(transformedFrame1_, transformedGrayFrame1_, CV_BGR2GRAY);
389 
390  // warp mask
391 
392  if (motionModel_ != MM_HOMOGRAPHY)
393  warpAffine(
394  mask1_, transformedMask1_, motion1to0(Rect(0,0,3,2)), mask1_.size(),
395  INTER_NEAREST);
396  else
397  warpPerspective(mask1_, transformedMask1_, motion1to0, mask1_.size(), INTER_NEAREST);
398 
399  erode(transformedMask1_, transformedMask1_, Mat());
400 
401  // update flow
402 
403  optFlowEstimator_->run(grayFrame_, transformedGrayFrame1_, flowX_, flowY_, flowErrors_);
404 
405  calcFlowMask(
406  flowX_, flowY_, flowErrors_, flowErrorThreshold_, mask, transformedMask1_,
407  flowMask_);
408 
409  body.flowX = flowX_;
410  body.flowY = flowY_;
411  body.mask0 = flowMask_;
412  body.mask1 = transformedMask1_;
413  body.frame1 = transformedFrame1_;
414  fmm_.run(flowMask_, body);
415 
416  completeFrameAccordingToFlow(
417  flowMask_, flowX_, flowY_, transformedFrame1_, transformedMask1_, distThresh_,
418  frame, mask);
419  }

481  CV_Assert(errors.type() == CV_32F && errors.size() == mask0.size());
482  CV_Assert(mask0.type() == CV_8U);
483  CV_Assert(mask1.type() == CV_8U && mask1.size() == mask0.size());

509 void completeFrameAccordingToFlow(

511  float distThresh, Mat &frame0, Mat &mask0)
512 {
513  CV_Assert(flowMask.type() == CV_8U);
514  CV_Assert(flowX.type() == CV_32F && flowX.size() == flowMask.size());

519  CV_Assert(mask0.type() == CV_8U && mask0.size() == flowMask.size());
520 
521  Mat_<uchar> flowMask_(flowMask), mask1_(mask1), mask0_(mask0);
522  Mat_<float> flowX_(flowX), flowY_(flowY);
523 
524  for (int y0 = 0; y0 < frame0.rows; ++y0)
525  {
526  for (int x0 = 0; x0 < frame0.cols; ++x0)
527  {
528  if (!mask0_(y0,x0) && flowMask_(y0,x0))
529  {
530  int x1 = cvRound(x0 + flowX_(y0,x0));
531  int y1 = cvRound(y0 + flowY_(y0,x0));
532 
533  if (x1 >= 0 && x1 < frame1.cols && y1 >= 0 && y1 < frame1.rows && mask1_(y1,x1)
534  && sqr(flowX_(y0,x0)) + sqr(flowY_(y0,x0)) < sqr(distThresh))
